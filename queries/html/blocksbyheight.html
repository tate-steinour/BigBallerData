Query

  SELECT DISTINCT t_name, ps_season, ps_name, ps_blck, p_height
  FROM player_stats
      JOIN team_stats ON ps_teamid=ts_id
      JOIN team ON t_id=ts_id
      JOIN player ON ps_name=p_name
  WHERE ps_blck >= 1
  ORDER BY ps_blck DESC
  LIMIT 10;
